        {\\ \quad\quad \Let{\; y \;}{\; \App{\code{is\_even}}{x'} \;}
        {\\ \quad\quad \App{\mathrm{not}}{y} \;}}} \;}
      }
  \end{array}, \\\\ &
    \Reduce{P}{\App{\code{is\_even}}{2} \\}{
    \Reduce{P}{
      & \Let{\; \code{cond} \;}{\; 2 = 0 \;}{ \\
      & \If{\; \code{cond} \;}{\; \true \\ &}{\\
      & \quad \Let{\; x' \;}{\; 2 - 1 \;}{\\
      & \quad \Let{\; y \;}{\App{\; \code{is\_odd}}{x' \;}}{\\
      & \quad \App{\mathrm{not}}{y}}}}} \\ }{
    \Reduce{P}{
      & \Let{\; \code{cond} \;}{\; \false \;}{ \\
      & \If{\; \code{cond} \;}{\; \true \\ &}{\\
      & \quad \Let{\; x' \;}{\; 2 - 1 \;}{\\
      & \quad \Let{\; y \;}{\App{\; \code{is\_odd}}{x' \;}}{\\
      & \quad \App{\mathrm{not}}{y}}}}} \\ }{
    \Reduce{P}{
      & \If{\; \false \;}{\; \true \\ &}{\\
      & \quad \Let{\; x' \;}{\; 2 - 1 \;}{\\
      & \quad \Let{\; y \;}{\App{\; \code{is\_odd}}{x' \;}}{\\
      & \quad \App{\mathrm{not}}{y}}}} \\ }{
    \Reduce{P}{
      & \Let{\; x' \;}{\; 2 - 1 \;}{\\
      & \Let{\; y \;}{\App{\; \code{is\_odd}}{x' \;}}{\\
      & \App{\mathrm{not}}{y}}} \\ }{
    \Reduce{P}{
      & \Let{\; x' \;}{\; 1 \;}{\\
      & \Let{\; y \;}{\App{\; \code{is\_odd}}{x' \;}}{\\
      & \App{\mathrm{not}}{y}}} \\ }{
    \Reduce{P}{
      & \Let{\; y \;}{\App{\; \code{is\_odd}}{1 \;}}{\\

      & \Let{\; y \;}{ \\

      & \quad\quad \Let{\; y}{\App{\; \code{is\_even}}{\; x' \;}}{ \\
      & \quad\quad \App{\mathrm{not}}{y\; }}}}}}{\\
      & \App{\mathrm{not}}{y}} \\ }{

        {\Reduce{P}{2 = 0}{\false}}

  & \Reduce{P}
        {\If{\false}{\true}{\App{\mathrm{not}}{\paren{\App{\code{is\_odd}}{\paren{2-1}}}}}}

        \sem{-}\paren{2, 1} = 1
